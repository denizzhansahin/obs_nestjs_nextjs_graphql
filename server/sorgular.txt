mutation updateEnrollment(
  $id: Float!, 
  $course_id: Float, 
  $student_id: Float, 
  $enrollment_date: DateTime, 
  $status: String, 
  $academician_id: Int
) {
  updateEnrollments(
    id: $id, 
    updateEnrollmentsData: {
      course_id: $course_id,
      student_id: $student_id,
      enrollment_date: $enrollment_date,
      status: $status,
      academician_id: $academician_id
    }
  ) {
    id
    enrollment_date
    status
    course {
      id
      name
    }
    students {
      userId
      first_name
      last_name
    }

    }
  }
}
